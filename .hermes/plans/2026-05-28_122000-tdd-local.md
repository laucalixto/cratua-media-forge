## Fase 1: Testes Rust em 30 minutos (máximo valor)

### 1.1 Testes inline no `ffmpeg.rs`
**Arquivo:** `crates/mediaforge-core/src/ffmpeg.rs` (adicionar `#[cfg(test)] mod tests` ao final)

Testes que rodam com `cargo test -p mediaforge-core` em < 1s:

```rust
#[cfg(test)]
mod tests {
    use super::*;
    use crate::job::EncodeParams;

    fn cmd_str(params: &EncodeParams) -> String {
        let cmd = build_command(params, Path::new("in.mp4"), Path::new("out.mp4"));
        format!("{:?}", cmd)
    }

    #[test]
    fn h264_codec() {
        let p = EncodeParams { video_codec: VideoCodec::H264, ..Default::default() };
        assert!(cmd_str(&p).contains("libx264"));
    }

    #[test]
    fn crf_present() {
        let p = EncodeParams { crf: Some(23), ..Default::default() };
        assert!(cmd_str(&p).contains("-crf"));
        assert!(cmd_str(&p).contains("23"));
    }

    #[test]
    fn bitrate_present() {
        let p = EncodeParams { video_bitrate: Some(5000), ..Default::default() };
        assert!(cmd_str(&p).contains("-b:v"));
        assert!(cmd_str(&p).contains("5000k"));
    }

    #[test]
    fn scale_in_vf() {
        let p = EncodeParams { width: 640, height: 360, ..Default::default() };
        assert!(cmd_str(&p).contains("scale=640:360"));
    }

    #[test]
    fn deinterlace_in_vf() {
        let p = EncodeParams { deinterlace: Some(DeinterlaceMethod::Yadif), ..Default::default() };
        assert!(cmd_str(&p).contains("yadif"));
    }

    #[test]
    fn no_threads_flag_when_zero() {
        let p = EncodeParams { threads: 0, ..Default::default() };
        assert!(!cmd_str(&p).contains("-threads"));
    }

    #[test]
    fn fps_fixed_adds_r_flag() {
        let p = EncodeParams { fps: FpsMode::Fixed(30), ..Default::default() };
        assert!(cmd_str(&p).contains("-r"));
        assert!(cmd_str(&p).contains("30"));
    }

    #[test]
    fn fps_same_no_r_flag() {
        let p = EncodeParams { fps: FpsMode::SameAsSource, ..Default::default() };
        assert!(!cmd_str(&p).contains("-r"));
    }

    #[test]
    fn trim_flags() {
        let p = EncodeParams {
            trim_start: Some("00:01:00".into()),
            trim_end: Some("00:05:00".into()),
            ..Default::default()
        };
        let c = cmd_str(&p);
        assert!(c.contains("-ss") && c.contains("00:01:00"));
        assert!(c.contains("-to") && c.contains("00:05:00"));
    }

    #[test]
    fn movflags_faststart() {
        let p = EncodeParams { movflags: vec![MovFlag::FastStart], ..Default::default() };
        assert!(cmd_str(&p).contains("-movflags"));
        assert!(cmd_str(&p).contains("faststart"));
    }

    #[test]
    fn audio_bitrate() {
        let p = EncodeParams { audio_bitrate: 192, ..Default::default() };
        assert!(cmd_str(&p).contains("-b:a"));
        assert!(cmd_str(&p).contains("192k"));
    }

    #[test]
    fn container_mp4() {
        let p = EncodeParams::default();
        assert!(cmd_str(&p).ends_with("out.mp4\""));
    }
}
```

## Fase 2: Testes JS em 20 minutos

### 2.1 Criar `test.js` standalone
**Arquivo:** `crates/mediaforge-tauri/ui/test.js`

Usa `node --test` (built-in do Node 18+):

```js
import { describe, it } from 'node:test';
import assert from 'node:assert/strict';

// ── Funções extraídas para teste ──
function even(n) {
    n = parseInt(n);
    if (isNaN(n)) return 1920;
    return n % 2 !== 0 ? n + 1 : n;
}

function normPath(p) {
    return (p || '').replace(/\\/g, '/');
}

function parseVideoFilter(s) {
    if (s === 'HFlip') return { HFlip: null };
    if (s === 'VFlip') return { VFlip: null };
    const m = s.match(/^Rotate\((\d+)\)$/);
    if (m) return { Rotate: parseInt(m[1]) };
    return { HFlip: null };
}

// ── Testes ──
describe('even()', () => {
    it('retorna par como está', () => assert.equal(even(360), 360));
    it('arredonda ímpar para cima', () => assert.equal(even(361), 362));
    it('NaN retorna 1920', () => assert.equal(even('abc'), 1920));
});

describe('normPath()', () => {
    it('converte backslash para slash', () => assert.equal(normPath('W:\\pasta\\file.mp4'), 'W:/pasta/file.mp4'));
    it('mantém slash', () => assert.equal(normPath('/home/user/file.mp4'), '/home/user/file.mp4'));
});

describe('parseVideoFilter()', () => {
    it('HFlip', () => assert.deepEqual(parseVideoFilter('HFlip'), { HFlip: null }));
    it('VFlip', () => assert.deepEqual(parseVideoFilter('VFlip'), { VFlip: null }));
    it('Rotate(90)', () => assert.deepEqual(parseVideoFilter('Rotate(90)'), { Rotate: 90 }));
});
```

Rodar: `node --test crates/mediaforge-tauri/ui/test.js`
